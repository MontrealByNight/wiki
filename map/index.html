// 1. Define your Themes (Basemaps) var darkTheme = L.tileLayer('https://{s}.basemaps.cartocdn.com/dark_all/{z}/{x}/{y}{r}.png', { attribution: '© CARTO', subdomains: 'abcd', maxZoom: 19 }); var lightTheme = L.tileLayer('https://{s}.basemaps.cartocdn.com/light_all/{z}/{x}/{y}{r}.png', { attribution: '© CARTO', subdomains: 'abcd', maxZoom: 19 }); var satelliteTheme = L.tileLayer('https://server.arcgisonline.com/ArcGIS/rest/services/World_Imagery/MapServer/tile/{z}/{y}/{x}', { attribution: 'Tiles © Esri', maxZoom: 19 }); // 2. Initialize the Map var map = L.map('map', { center: [45.53, -73.60], zoom: 11, layers: [darkTheme] // Default theme }); // 3. Create the Empty Overlay Groups (These are the checkboxes!) var territoryLayer = L.layerGroup().addTo(map); var camarillaLayer = L.layerGroup().addTo(map); var anarchLayer = L.layerGroup().addTo(map); var contestedLayer = L.layerGroup().addTo(map); // 4. Define Colors function getColor(type) { if (type === 'Camarilla') return '#e67e22'; // Orange if (type === 'Anarch') return '#206694'; // Blue if (type === 'Contested' || type === 'Hecata') return '#992d22'; // Dark Red return '#546e7a'; // Grey } // 5. Draw Polygons into the Territory Layer if (typeof mapPolygons !== 'undefined') { mapPolygons.forEach(function(poly) { var zoneColor = getColor(poly.type); L.polygon(poly.coords, { color: zoneColor, weight: 2, fillColor: zoneColor, fillOpacity: 0.15 }) .bindPopup("" + poly.name + "
Type: " + poly.type) .addTo(territoryLayer); // Notice it goes to territoryLayer, not 'map' }); } // 6. Helper Function to Draw Pins into Specific Layers function addPinsToLayer(dataArray, targetLayer) { if (typeof dataArray !== 'undefined') { dataArray.forEach(function(loc) { var color = getColor(loc.type); L.circleMarker([loc.lat, loc.lng], { color: color, fillColor: color, fillOpacity: 0.8, radius: 7 }) .bindPopup("" + loc.name + "
" + (loc.desc ? loc.desc : "")) .addTo(targetLayer); }); } } // Fill the layers with your data addPinsToLayer(typeof data_camarilla !== 'undefined' ? data_camarilla : [], camarillaLayer); addPinsToLayer(typeof data_anarch !== 'undefined' ? data_anarch : [], anarchLayer); addPinsToLayer(typeof data_contested !== 'undefined' ? data_contested : [], contestedLayer); // 7. Create the Unified Menu var baseMaps = { "Dark Theme": darkTheme, "Light Theme": lightTheme, "Satellite View": satelliteTheme }; var overlayMaps = { "Territories": territoryLayer, "Camarilla Pins": camarillaLayer, "Anarch Pins": anarchLayer, "Contested/Other Pins": contestedLayer }; // Add the menu to the top right L.control.layers(baseMaps, overlayMaps, { position: 'topright' }).addTo(map);
